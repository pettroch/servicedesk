# Создать заявку +
POST /api/task/create
body {
    name: str,
    description: str,
    status: id,
    date: str,
    files: [], -
    user_id: id,
    contractor_id: id
}

return {
    task_id: int
}



# Получить заявку по айди +
GET /api/task/get
body {
    task_id: int
}

return {
    name: str,
    description: str,
    status: id,
    date: str,
    files: [], -
    user_id: id,
    contractor_id: id
}
